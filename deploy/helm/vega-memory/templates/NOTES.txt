1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  export HOSTNAME={{ (index .Values.ingress.hosts 0).host }}
  echo "http://$HOSTNAME/"
{{- else if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward svc/{{ include "vega-memory.fullname" . }} 3000:{{ .Values.service.port }}
  echo "http://127.0.0.1:3000/api/health"
{{- else }}
  kubectl get svc {{ include "vega-memory.fullname" . }}
{{- end }}

2. Export the API key from the configured Secret:
  export VEGA_API_KEY=$(kubectl get secret {{ .Values.existingSecret | default (printf "%s-secrets" (include "vega-memory.fullname" .)) }} -o jsonpath='{.data.{{ .Values.existingSecretKey }}}' | base64 --decode)

3. Verify the API is healthy:
  curl -fsS -H "Authorization: Bearer ${VEGA_API_KEY}" http://127.0.0.1:3000/api/health

4. Run the Helm test:
  helm test {{ .Release.Name }}

5. Supply a strong API key before install or upgrade:
  helm upgrade {{ .Release.Name }} {{ .Chart.Name }} --set secret.apiKey='<strong-random-value>'
